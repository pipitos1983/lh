#!/bin/bash


# chkconfig: 35 54 34

# description: Starts and stops the db2 instance


# Default-Start: 2 3 4 5

# Default-Stop: 0 1 6



if [ -f /home/dasusr1/das/dasprofile ]; then

. /home/dasusr1/das/dasprofile

fi



PATH="/opt/ibm/db2/V10.1/das/bin:$PATH"


if [ -f /home/db2inst1/sqllib/db2profile ]; then

. /home/db2inst1/sqllib/db2profile

fi




function start() {

su db2inst1 -c "db2start"

su dasusr1 -c "db2admin start"

}



function stop() {

su db2inst1 -c "db2stop"

su dasusr1 -c "db2admin stop"

}


function restart() {

stop

start

}


function status() {

su db2inst1 -c "db2ls"

}



function usage() {

echo "$0 start|stop|restart|status"

}


case $1 in

'')

usage

exit 1

;;

restart)

restart

;;

start)

start

;;

status)

status

;;

stop)

stop

;;

*)

usage

exit 1

;;

esac
