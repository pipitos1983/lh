Для работы в виртуалке:


# pacman -S open-vm-tools

# systemctl enable vmtoolsd

# systemctl start vmtoolsd


Сеть:

в общем случае


# pacman -S netctl

# cp /etc/netctl/examples/ethernet-dhcp /etc/netctl/eth-dhcp

# vi /etc/netctl/eth-dhcp

вносим разумные правки (как правило имя интерфейса)


# netctl start eth-dhcp

# netctl enable eth-dhcp



Локализация консоли:


pacman -S kbd-ru-keymaps

pacman -S terminus-font


/etc/vconsole.conf:

LOCALE="ru_RU.UTF-8"

KEYMAP="ru"

FONT="ter-v16v"

CONSOLEMAP=""


/ect/locale.conf:

LANG=ru_RU.UTF-8

LC_MESSAGES=ru_RU.UTF-8


/etc/mkinitcpio.conf:

HOOKS="base consolefont keymap ..."


mkinitcpio -p linux


reboot
