fedora24+ - проблем не выявлено, но 32битная версия 25-й федоры сама по себе виснет и для установки на старый комп для испльзования в роли хаба не подходит. (проверено только на виртуалках)


debian 8.7(i386):


сервер:


aptitude install usbip # установка утилит


modprobe usbip-core # загрузка

modprobe usbip-host # модулей


usbipd -D #запуск демона


usbip list -l # просмотр локальных устройств доступных для "расшаривания".


usbip bind -b 2-1 # "расширивание" устройства.


клиент:


usbip list -r адрес.сервера # просмотр доступных для подключения удаленных юсб устройств


{

пример:


Exportable USB devices

======================

- 10.200.0.65

2-1: Hewlett-Packard : unknown product (03f0:3407)

: /sys/devices/pci0000:00/0000:00:11.0/0000:02:02.0/usb2/2-1

: (Defined at Interface level) (00/00/00)

}


usbip attach -r адрес.сервера -b 2-1 # подключение удаленного устройства с busid 2-1 (см. пример выше)


usbip -port #просмотр подключенных устройств

usbip detach --port 0 #отключение устройства 0
