предположим добавляем диск da2 120GB

в пулл под названием mir1


gpart create -s gpt da2 # создание таблицы разделов

gpart add -s 16G -t freebsd-zfs -a 4k da2 # добавление раздела на 16ГБ (da2p1)

gpart add -t freebsd-zfs -a 4k da2 # добавление раздела на все оставшееся пространство (da2p2)

zpool add mir1 log da2p1 # добавление в пулл раздела под логи.

zpool add mir1 cache da2p2 # добавление в пулл раздела под кеш.
